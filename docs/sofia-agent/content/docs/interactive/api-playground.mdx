## Available Endpoints

The playground provides access to the following API endpoints:

### Agent Management

- **Create Agent**: Create a new SOFIA agent with configuration
- **List Agents**: Get a list of available agents
- **Get Agent**: Retrieve agent details and configuration
- **Update Agent**: Modify an existing agent
- **Delete Agent**: Remove an agent

### Session Operations

- **Create Session**: Start a new conversation session
- **Get Session**: Retrieve session details and history
- **Send Message**: Send a user message and get agent response
- **End Session**: Terminate an active session

### Tool Integration

- **List Available Tools**: Get all available tools
- **Register Custom Tool**: Add a new tool to the system
- **Invoke Tool Directly**: Call a tool with specified parameters

      ```go
      package main

      import (
        "bytes"
        "encoding/json"
        "fmt"
        "net/http"
      )

      func main() {
        // Create agent configuration
        type Route struct {
          When   string `json:"when,omitempty"`
          Goto   string `json:"goto,omitempty"`
          Default string `json:"default,omitempty"`
        }

        type Step struct {
          ID     string  `json:"id"`
          Prompt string  `json:"prompt"`
          Next   string  `json:"next,omitempty"`
          Routes []Route `json:"routes,omitempty"`
        }

        type AgentConfig struct {
          Name        string `json:"name"`
          Description string `json:"description"`
          Steps       []Step `json:"steps"`
        }

        agent := AgentConfig{
          Name:        "Customer Support Agent",
          Description: "Handles customer inquiries and issues",
          Steps: []Step{
            {
              ID:     "greeting",
              Prompt: "Greet the customer and ask how you can help",
              Next:   "listen",
            },
            {
              ID:     "listen",
              Prompt: "Listen to the customer's issue",
              Routes: []Route{
                {When: "Customer has billing issue", Goto: "billing_issue"},
                {When: "Customer has technical issue", Goto: "technical_issue"},
                {Default: "general_help"},
              },
            },
            // Additional steps omitted for brevity
          },
        }

        // Convert to JSON
        jsonData, err := json.Marshal(agent)
        if err != nil {
          fmt.Println("Error:", err)
          return
        }

        // Create request
        req, err := http.NewRequest("POST", "https://api.example.com/agents", bytes.NewBuffer(jsonData))
        if err != nil {
          fmt.Println("Error:", err)
          return
        }

        // Add headers
        req.Header.Set("Authorization", "Bearer YOUR_API_KEY")
        req.Header.Set("Content-Type", "application/json")

        // Send request
        client := &http.Client{}
        resp, err := client.Do(req)
        if err != nil {
          fmt.Println("Error:", err)
          return
        }
        defer resp.Body.Close()

        // Process response
        var result map[string]interface{}
        json.NewDecoder(resp.Body).Decode(&result)
        fmt.Println(result)
      }
      ```

      ```java
      import java.net.URI;
      import java.net.http.HttpClient;
      import java.net.http.HttpRequest;
      import java.net.http.HttpResponse;
      import java.net.http.HttpResponse.BodyHandlers;
      import org.json.JSONArray;
      import org.json.JSONObject;

      public class SofiaApiExample {
          public static void main(String[] args) {
              try {
                  // Create the request body
                  JSONObject agentConfig = new JSONObject();
                  agentConfig.put("name", "Customer Support Agent");
                  agentConfig.put("description", "Handles customer inquiries and issues");
                  
                  JSONArray steps = new JSONArray();
                  
                  // Greeting step
                  JSONObject greeting = new JSONObject();
                  greeting.put("id", "greeting");
                  greeting.put("prompt", "Greet the customer and ask how you can help");
                  greeting.put("next", "listen");
                  steps.put(greeting);
                  
                  // Listen step
                  JSONObject listen = new JSONObject();
                  listen.put("id", "listen");
                  listen.put("prompt", "Listen to the customer's issue");
                  
                  JSONArray routes = new JSONArray();
                  
                  JSONObject billingRoute = new JSONObject();
                  billingRoute.put("when", "Customer has billing issue");
                  billingRoute.put("goto", "billing_issue");
                  routes.put(billingRoute);
                  
                  JSONObject technicalRoute = new JSONObject();
                  technicalRoute.put("when", "Customer has technical issue");
                  technicalRoute.put("goto", "technical_issue");
                  routes.put(technicalRoute);
                  
                  JSONObject defaultRoute = new JSONObject();
                  defaultRoute.put("default", "general_help");
                  routes.put(defaultRoute);
                  
                  listen.put("routes", routes);
                  steps.put(listen);
                  
                  // Add steps to config
                  agentConfig.put("steps", steps);
                  
                  // Create HTTP client
                  HttpClient client = HttpClient.newHttpClient();
                  
                  // Create request
                  HttpRequest request = HttpRequest.newBuilder()
                      .uri(URI.create("https://api.example.com/agents"))
                      .header("Authorization", "Bearer YOUR_API_KEY")
                      .header("Content-Type", "application/json")
                      .POST(HttpRequest.BodyPublishers.ofString(agentConfig.toString()))
                      .build();
                  
                  // Send request
                  HttpResponse<String> response = client.send(request, BodyHandlers.ofString());
                  
                  // Print response
                  System.out.println(response.statusCode());
                  System.out.println(response.body());
                  
              } catch (Exception e) {
                  e.printStackTrace();
              }
          }
      }
      ```
